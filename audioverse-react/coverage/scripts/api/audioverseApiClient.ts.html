// audioverseApiClient.ts
import axios from "axios";
 
/**
 * Jedno źródło prawdy dla hosta API.
 * Priorytet: Vite env -> CRA env -> window global -> domyślny.
 */
/*const DEFAULT_API_ROOT = "https://api.audioverse.io";*/
const DEFAULT_API_ROOT = "http://localhost:5000";
axios.defaults.withCredentials = true;
 
// bezpieczne pobranie Vite env (import.meta.env.*)
const fromVite = (() => {
    try {
        return import.meta.env.VITE_AUDIOVERSE_API_ROOT as string | undefined;
    } catch {
        return undefined;
    }
})();
 
// bez bezpośredniego użycia `process` (TS nie zna typu w przeglądarce)
const fromCra = ((globalThis as Record<string, unknown>)
    ?.process as { env?: Record<string, string | undefined> } | undefined)
    ?.env?.REACT_APP_AUDIOVERSE_API_ROOT;
 
// możliwość nadpisania w oknie (np. <script>window.__AUDIOVERSE_API_ROOT__="..."</script>)
const fromWindow = globalThis.__AUDIOVERSE_API_ROOT__;
 
export const API_ROOT: string = fromVite || fromCra || fromWindow || DEFAULT_API_ROOT;
 
// Główny klient axios
export const apiClient = axios.create({
    baseURL: API_ROOT,
    headers: { "Content-Type": "application/json" },
});
 
// Interceptor do dodawania Authorization headera z bieżącego tokena
// Import po utworzeniu apiClient aby uniknąć circular dependency
apiClient.interceptors.request.use((config) => {
    // Dynamically get the token at request time
    // This will be resolved by apiUser module
    const getToken = () => {
        try {
            const stored = localStorage.getItem("audioverse_access_token");
            if (stored) {
                return stored;
            }
        } catch (e) {
            // localStorage not available
        }
        return null;
    };
    
    const token = getToken();
    // console.log("🌐 Request interceptor - URL:", config.url, "Token:", token ? "YES (" + token.substring(0, 30) + "...)" : "NO");
    if (token) {
        const headerValue = `Bearer ${token}`;
        config.headers.Authorization = headerValue;
        // console.log("🌐 Authorization header ADDED:", headerValue.substring(0, 50) + "...");
        // console.log("🌐 Config headers after:", JSON.stringify(config.headers));
    } else {
        // console.log("🌐 No token, request will be sent WITHOUT Authorization");
    }
    return config;
}, (error) => {
    return Promise.reject(error);
});
 
// Response interceptor to log 401 errors
// Session timeout handler: auto logout and redirect on 401
// Rate limiting dla logowania błędów
const errorLogCache = new Map<string, number>();
const ERROR_LOG_COOLDOWN = 2000; // 2 sekundy między logowaniem tego samego błędu
 
apiClient.interceptors.response.use(
    (response) => response,
    (error) => {
        const errorKey = `${error.config?.url}-${error.response?.status}`;
        const now = Date.now();
        const lastLog = errorLogCache.get(errorKey) || 0;
        
        // Loguj tylko jeśli minęło wystarczająco czasu od ostatniego loga
        if (now - lastLog > ERROR_LOG_COOLDOWN) {
            errorLogCache.set(errorKey, now);
            
            if (error.response?.status === 401) {
                console.debug("❌ Got 401 Unauthorized for:", error.config?.url);
            } else if (error.code === 'ERR_INSUFFICIENT_RESOURCES' || error.response?.status === 503) {
                console.error("⚠️ Backend overloaded - rate limiting requests:", error.config?.url);
            }
        }
        
        if (error.response?.status === 401) {
            // Wyloguj użytkownika i przekieruj na /login
            try {
                localStorage.removeItem("audioverse_access_token");
                localStorage.removeItem("audioverse_refresh_token");
            } catch {}
            if (window.location.pathname !== "/login") {
                window.location.href = "/login";
            }
        }
        return Promise.reject(error);
    }
);
 
// Pomocnicza funkcja do budowy ścieżek modułów
export const apiPath = (moduleBase: string, path: string) =>
    `${moduleBase.replace(/\/$/, "")}/${path.replace(/^\//, "")}`;
 
// Stałe modułów
export const DMX_BASE = "/api/dmx";
export const EDITOR_BASE = "/api/editor";
 
// Stałe przejęte z libraryApiClient (Library API wcielone do AudioVerse API)
export const AUTH_BASE = "/api/auth";
export const PLAYLIST_BASE = "/api/library/playlists";
export const LIBRARY_BASE = "/api/library";
